001 import java.io.*;
002 import java.net.*;
003 import java.util.*;
004 import java.awt.*;
005 import javax.swing.*;
006 
007 public class MultiThreadServer extends JFrame {
008   // Text area for displaying contents
009   private JTextArea jta = new JTextArea();
010 
011   public static void main(String[] args) {
012     new MultiThreadServer();
013   }
014 
015   public MultiThreadServer() {
016     // Place text area on the frame
017     getContentPane().setLayout(new BorderLayout());
018     getContentPane().add(new JScrollPane(jta), BorderLayout.CENTER);
019 
020     setTitle("MultiThreadServer");
021     setSize(500, 300);
022     setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
023     setVisible(true); // It is necessary to show the frame here!
024 
025     try {
026       // Create a server socket
027       ServerSocket serverSocket = new ServerSocket(8000);
028       jta.append("MultiThreadServer started at " + new Date() + '\n');
029 
030       // Number a client
031       int clientNo = 1;
032 
033       while (true) {
034         // Listen for a new connection request
035         Socket socket = serverSocket.accept();
036 
037         // Display the client number
038         jta.append("Starting thread for client " + clientNo +
039           " at " + new Date() + '\n');
040 
041         // Find the client's host name, and IP address
042         InetAddress inetAddress = socket.getInetAddress();
043         jta.append("Client " + clientNo + "'s host name is "
044           + inetAddress.getHostName() + "\n");
045         jta.append("Client " + clientNo + "'s IP Address is "
046           + inetAddress.getHostAddress() + "\n");
047 
048         // Create a new thread for the connection
049         HandleAClient thread = new HandleAClient(socket);
050 
051         // Start the new thread
052         thread.start();
053 
054         // Increment clientNo
055         clientNo++;
056       }
057     }
058     catch(IOException ex) {
059       System.err.println(ex);
060     }
061   }
062 
063   // Inner class
064   // Define the thread class for handling new connection
065   class HandleAClient extends Thread {
066     private Socket socket; // A connected socket
067 
068     /** Construct a thread */
069     public HandleAClient(Socket socket) {
070       this.socket = socket;
071     }
072 
073     /** Run a thread */
074     public void run() {
075       try {
076         // Create data input and output streams
077         DataInputStream inputFromClient = new DataInputStream(
078           socket.getInputStream());
079         DataOutputStream outputToClient = new DataOutputStream(
080           socket.getOutputStream());
081 
082         // Continuously serve the client
083         while (true) {
084           // Receive radius from the client
085           double radius = inputFromClient.readDouble();
086 
087           // Compute area
088           double area = radius * radius * Math.PI;
089 
090           // Send area back to the client
091           outputToClient.writeDouble(area);
092 
093           jta.append("radius received from client: " +
094             radius + '\n');
095           jta.append("Area found: " + area + '\n');
096         }
097       }
098       catch(IOException e) {
099         System.err.println(e);
100       }
101     }
102   }
103 }
